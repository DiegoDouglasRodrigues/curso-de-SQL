criando tabela ----------------------------------

create table pessoas (
id int not null auto_increment,
nome varchar(30) not null,
nascimento date,
sexo enum('M', 'F'),
peso decimal(5,2),
altura decimal(3,2),
nacionalidade varchar(20) default 'Brasil',
primary key (id)
) DEFAULT CHARSET = UTF8;

describe pessoas;

create table if not exists cursos (
nome varchar(30) not null unique,
descricao text,
carga int unsigned,
totaulas int unsigned,
ano year default '2023'
) default charset = utf8;

describe cursos;

--- adicionado coluna ID
alter table cursos
add column idcurso int first;


alter table cursos

Aula 10 – PHPMyAdmin *





use cadastro;


--- descente
select * from cursos
order by nome;

-- descrecente
select * from cursos
order by nome desc;

--- selecoes parciais
select nome, carga, ano from cursos
order by nome;


--- selecoes ordenada
select ano, nome, carga from cursos
order by ano, nome;


----- filtrabdo linhas , pode usar colunas ou * para todos
select nome, carga from cursos 
where ano = '2016'
order by nome;


----- filtrando linhas , pode usar colunas ou * para todos
select nome, descricao, ano from cursos 
where ano <= '2016'
order by ano, nome;


----- filtrando linhas beetewen ( entre)
select nome,  ano from cursos 
where ano between '2014' and '2016'
order by ano, nome;


----- filtrando linhas beetewen ( IN so mostra o que esta no parametro, no caso 2014 e 2016 'nao exibe 2015')
select nome,  ano from cursos 
where ano in (2014, 2016)
order by ano, nome;


select nome,  ano, descricao from cursos 
where carga > 35 and totaulas < 30;
